\noindent\textit{Attempt each question before checking answers. Time yourself. No calculators.}

\section*{Section A \quad Rapid Recognition \hfill{\normalfont\small($\leq$15\,s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Prime factorisation, divisibility rules, and modular arithmetic basics.}

\begin{enumerate}

\item Find the largest prime factor of $2025$.

\item What is the last digit of $3^{2026}$?

\item Find the greatest common divisor of $84$ and $120$.

\item What is the remainder when $10^{10}$ is divided by $9$?

\item Find the missing digit $d$ such that the five-digit number $47d53$ is divisible by $9$.

\item How many positive integer factors does $100$ have?

\item Which is larger: $2^{300}$ or $3^{200}$?

\item Evaluate $5^{2024} \pmod 4$.

\item Find the smallest positive integer $n$ such that $12n$ is a perfect square.

\item What is the sum of the prime factors of $210$?

\end{enumerate}

% ══════════════════════════════════════════════════════════════════════════════
\section*{Section B \quad Manipulation Drills \hfill{\normalfont\small($\leq$50\,s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Linear congruences, powers of primes, and prime structure.}

\begin{enumerate}

\item Find the number of trailing zeroes of $20!$.

\item Solve for $x$ in $3x \equiv 4 \pmod 7$, with $0 < x < 7$.

\item If $p$ and $q$ are prime numbers such that $p^2 - q^2 = 24$, find the value of $p+q$.

\item Find the last two digits of $5^{2025}$.

\item A number $N$ leaves a remainder of $2$ when divided by $3$, and $3$ when divided by $5$. What is the smallest positive integer value of $N$?

\item How many perfect squares divide $3600$?

\item Find the highest power of $3$ dividing $27^5 \times 9^4$.

\item What is the smallest positive integer with exactly $6$ positive divisors?

\item If $x$ and $y$ are positive integers such that $xy = 144$ and $\gcd(x,y)=3$, find the maximum possible value of $x+y$.

\item Calculate the remainder when $1! + 2! + 3! + \dots + 10!$ is divided by $5$.

\end{enumerate}

% ══════════════════════════════════════════════════════════════════════════════
\section*{Section C \quad Substitution \& Structure \hfill{\normalfont\small($\leq$90\,s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Synoptic links: use algebra and combinatorics alongside Diophantine logic.}

\begin{enumerate}

\item Find the number of pairs of positive integers $(x, y)$ such that $xy + x + y = 23$. \textit{\small(after SMC 2018 Q16)}

\item Find the sum of all prime numbers $p$ such that $p^2 + 2$ is also prime. \textit{\small(after SMC 2016 Q20)}

\item How many 3-digit palindromes are divisible by $11$? \textit{\small(after SMC 2015 Q19)}

\item The product of three consecutive positive integers is equal to $336$. Find the sum of these integers. \textit{\small(after SMC 2012 Q14)}

\item Let $N$ be a 4-digit number. When its digits are reversed, the new number is $N + 1089$. If the sum of the digits of $N$ is $12$, what is the maximum possible value of $N$? \textit{\small(after SMC 2017 Q21)}

\item Find the sum of all positive integers $x$ such that $\frac{x+11}{x-3}$ is an integer. \textit{\small(after SMC 2019 Q18)}

\item Find the maximum possible value of $\gcd(7n+15, 3n+2)$, where $n$ is a positive integer. \textit{\small(after SMC 2013 Q21)}

\item Find the sum of all integers $n$ such that $\frac{n^2+3n+5}{n-1}$ is an integer. \textit{\small(after BMO1 2007 Q1)}

\end{enumerate}

% ══════════════════════════════════════════════════════════════════════════════
\section*{Section D \quad Challenge \hfill{\normalfont\small(BMO1 difficulty)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Insight over computation. Uncover the structure.}

\begin{enumerate}

\item Find all three-digit numbers $abc$ such that $abc = a! + b! + c!$. \textit{\small(after SMC 2004 Q21)}

\item Let $d_n = \gcd(n^2 + 3, n+1)$. Find the sum of all possible values of $d_n$ as $n$ ranges over all positive integers. \textit{\small(after SMC 2010 Q24)}

\item Find the last three digits of $2025^{2025}$. \textit{\small(after SMC 2008 Q24)}

\item For how many positive integers $n \le 100$ is $n^n$ a perfect square? \textit{\small(after SMC 2001 Q22)}

\item Find the maximum value of $x + y + z$ given that $x, y, z$ are positive integers satisfying $x \le y \le z$ and $\frac{1}{x} + \frac{1}{y} + \frac{1}{z} = 1$. \textit{\small(after BMO1 2011 Q1)}

\end{enumerate}
