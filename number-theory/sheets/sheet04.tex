\section*{Section A \quad Rapid Recognition \hfill{\normalfont\small($\leq$15 s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Prime factors, divisibility rules, and quick remainders.}

\begin{enumerate}

\item Find the largest prime factor of $1001$.

\item What is the last digit of $2^{2026}$?

\item What is the remainder when $2^{10} + 3^{10}$ is divided by $5$?

\item Find the greatest common divisor of $120$ and $84$.

\item How many positive integer factors does $144$ have?

\item What is the remainder when $11 \times 12 \times 13$ is divided by $10$?

\item What is the smallest positive integer divisible by $2, 3, 4, 5,$ and $6$?

\item Find the sum of the prime factors of $210$.

\item What is the remainder when $10^{2026} - 1$ is divided by $9$?

\item Find the value of $k$ if $2^k$ is the highest power of $2$ dividing $20!$.

\end{enumerate}

\section*{Section B \quad Manipulation Drills \hfill{\normalfont\small($\leq$50 s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Modulo arithmetic properties, CRT basics, and divisor counting.}

\begin{enumerate}

\item Find the smallest positive integer $n$ such that $15n$ is a perfect square.

\item Find all single digits $d$ such that the four-digit number $5d72$ is divisible by $9$.

\item Find the number of trailing zeros in $50!$.

\item What is the remainder when $5^{2025}$ is divided by $7$?

\item If $x \equiv 3 \pmod 7$, find the remainder when $x^2 + 4x + 5$ is divided by $7$.

\item Find the smallest positive integer that leaves a remainder of $2$ when divided by $3$, and $3$ when divided by $5$.

\item Find the sum of all positive integer divisors of $100$.

\item What is the last digit of $3^{101} + 7^{101}$?

\item Find the number of positive divisors of $2^3 \times 3^4 \times 5$.

\item Find the largest three-digit number that is a multiple of $11$ and $13$.

\end{enumerate}

\section*{Section C \quad Substitution \& Structure \hfill{\normalfont\small($\leq$90 s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Look for algebraic structures like difference of squares and combinatorial counting strategies.}

\begin{enumerate}

\item Find all pairs of positive integers $(x, y)$ such that $x^2 - y^2 = 45$. \textit{\small(after SMC 2018 Q15)}

\item The prime number $p$ can be expressed as $p = a^2 - b^2$, where $a$ and $b$ are also prime numbers. Find the value of $p$. \textit{\small(after SMC 2011 Q19)}

\item Find the largest integer $n$ such that $n^3 + 100$ is divisible by $n+10$. \textit{\small(after MAT 2016 Q1H)}

\item How many ordered pairs of positive integers $(x, y)$ satisfy $\frac{1}{x} + \frac{1}{y} = \frac{1}{6}$? \textit{\small(after SMC 2015 Q21)}

\item Find all integer values of $x$ for which $\frac{x^2 + 5x + 14}{x+3}$ is an integer. \textit{\small(after SMC 2019 Q20)}

\item How many divisors of $10^{10}$ are perfect squares? \textit{\small(after SMC 2021 Q22)}

\item How many positive integers $n \leq 1000$ are not divisible by $2, 3,$ or $5$? \textit{\small(after SMC 2017 Q19)}

\item Given that $p$ and $q$ are prime numbers satisfying $p = q^2 - 36$, find the value of $p$. \textit{\small(after SMC 2014 Q22)}

\end{enumerate}

\section*{Section D \quad Challenge \hfill{\normalfont\small(BMO1 difficulty)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Insight over computation. Each question has a short, elegant solution --- find it.}

\begin{enumerate}

\item Find all pairs of positive integers $(m, n)$ such that $3^m - 2^n = 1$. \textit{\small(after BMO1 2008 Q1)}

\item For how many positive integers $n$ is $n^4 + 4^n$ a prime number? \textit{\small(after BMO1 2013 Q2)}

\item A sequence is defined by $a_1 = 1$, $a_2 = 1$, and $a_{n+2} = a_{n+1} + a_n$. What is the remainder when $a_{2026}$ is divided by $3$? \textit{\small(after SMC 2022 Q25)}

\item Let $p, q,$ and $r$ be prime numbers such that $p = 9q^2 - r^2$. Find the value of $p$. \textit{\small(after BMO1 2010 Q1)}

\item Find the largest positive integer $n$ such that $2^n$ divides $3^{1024} - 1$. \textit{\small(after BMO1 2019 Q3)}

\end{enumerate}
