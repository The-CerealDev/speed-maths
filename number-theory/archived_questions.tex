\section*{Archived Number Theory Questions}
\begin{enumerate}
% From Sheet 01

\item Find the number of positive integers $n \leq 100$ such that $n^2 - n$ is divisible by $10$.

\item A rectangular floor of dimensions $20 \times 25$ is covered by $1 \times 1$ tiles. A straight line is drawn from one corner to the opposite corner. How many tiles does it pass through?

\item Find the product of all integers $n$ such that $n^2 + 2n + 13$ is a perfect square.

% From Sheet 02

\item Which of the numbers $5, 7, 11,$ or $13$ is a factor of $2^{2024} + 2^{2023} + 2^{2022}$? \textit{\small(after SMC 2023 Q10)}

\item How many subsets of size 2 from $\{1, 2, \dots, 20\}$ have a sum that is an even number? \textit{\small(after SMC 2020 Q15)}

\item Find the sum of all integers $x$ for which $x^4 - 2x^2 + 1$ is a prime number. \textit{\small(after MAT 2021 Q1)}

% From Sheet 03

\item Find the sum of the prime factors of $2^{16} - 1$. \textit{\small(after SMC substitution)}

\item The number $N = 1! \times 2! \times 3! \times \cdots \times 9! \times 10!$ can be written as $k \times m^2$, where $k$ and $m$ are positive integers and $k$ is square-free. Find $k$. \textit{\small(after SMC prime powers)}

\item Let $P(x)$ be a polynomial with integer coefficients such that $P(1) = 2025$ and $P(5) = 2026$. Find the number of integer roots of $P(x)$. \textit{\small(after SMC polynomials)}

% From Sheet 04

\item Given that $p$ and $q$ are prime numbers such that $p^2 - 4q^2 = 21$, find $p$ and $q$. \textit{\small(after BMO1 2011 Q1)}

\item The roots of the quadratic equation $x^2 - px + q = 0$ are positive integers. If $p$ and $q$ are prime numbers, find $p$ and $q$. \textit{\small(after SMC 2020 Q24)}

\item Find the only prime number $p$ such that $p^2 - p + 1$ is a perfect cube. \textit{\small(after BMO1 2017 Q1)}

% From Sheet 06

\item Find the number of integer solutions $(x,y)$ to the equation $\frac{1}{x} + \frac{1}{y} = \frac{1}{3}$. \textit{\small(after BMO1 2010 Q1)}

\item Find the sum of all two-digit numbers that are exactly $3$ times the product of their digits. \textit{\small(after SMC 2014 Q15)}

\item Find the sum of all positive integers $n$ such that $n^2 + 20n + 11$ is a perfect square. \textit{\small(after BMO1 2018 Q1)}
\end{enumerate}
