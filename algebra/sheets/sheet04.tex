\SpeedTitleBlock{Daily Algebra Drill \#4}
\noindent\textit{New toolkit today: Geometric series identity $\cdot$ Polynomial remainder shortcut $\cdot$ Parity and mod arguments $\cdot$ AM--GM under substitution.}

\vspace{2pt}
\noindent\textit{Attempt each question before checking answers. Time yourself. No calculators.}

\section*{Section A \quad Rapid Recognition \hfill{\normalfont\small($\leq$15 s each)}}

\noindent\textit{No expansion. Recognise the pattern, write the answer.}

\begin{enumerate}

\item Evaluate without expanding: $\;201 \times 199$.

\item Factorise: $\;x^2 - x - 42$.

\item Simplify: $\;1 + x + x^2 + x^3 + x^4$ when $x = -1$.

\item Given $p + q = 6$ and $pq = 7$, find $p^3 + q^3$.

\item Without long division, find the remainder when $f(x)=x^3+2x-3$ is divided by $(x-1)$.

\item Factorise completely: $\;2x^3 - 18x$.

\item Simplify: $\;\dfrac{3^{n+1} - 3^{n-1}}{3^n}$.

\item Given $a - b = 3$ and $a^2 + b^2 = 29$, find $ab$.

\item Factorise: $\;x^4 - 1$ completely over $\mathbb{R}$.

\item Simplify: $\;\dfrac{n!}{(n-2)!}$.

\end{enumerate}

\section*{Section B \quad Manipulation Drills \hfill{\normalfont\small($\leq$50 s each)}}

\noindent\textit{Fractions, structure, identities. Minimum working.}

\begin{enumerate}

\item Using the geometric series formula, simplify: $\;\dfrac{x^5 - 1}{x - 1}$.

\item Find the remainder when $g(x) = x^{10} - 3x^5 + 2$ is divided by $(x-1)$
      and by $(x+1)$.

\item Simplify: $\;\dfrac{ab}{a-b} - \dfrac{a^2b}{a^2-b^2}$.

\item Factorise: $\;x^6 - y^6$ completely over $\mathbb{Z}$.

\item The roots of $3x^2 + 5x - 2 = 0$ are $\alpha$ and $\beta$.
      Find $\dfrac{1}{\alpha} + \dfrac{1}{\beta}$.

\item Simplify: $\;\left(1 - \dfrac{1}{x^2}\right)\!\left(x^2 + x\right)$.

\item Given $a + b + c = 5$, $\;ab + bc + ca = 6$, find $a^2 + b^2 + c^2$.

\item Rearrange for $y$: $\;\dfrac{2y+1}{y-3} = x$.

\item Show that $\;4n^2 - 1 = (2n-1)(2n+1)$, hence prove that $4n^2 - 1$
      is never prime for integers $n \geq 2$.

\item Given $\alpha, \beta$ satisfy $\alpha + \beta = 4$ and $\alpha\beta = 1$,
      find a quadratic with integer coefficients whose roots are
      $\alpha - \tfrac{1}{\alpha}$ and $\beta - \tfrac{1}{\beta}$.

\end{enumerate}

\section*{Section C \quad Substitution \& Structure \hfill{\normalfont\small($\leq$90 s each)}}

\noindent\textit{One clean substitution is worth ten lines of algebra.}

\begin{enumerate}

\item Solve: $\;4^x - 5 \cdot 2^x + 4 = 0$.

\item Given $t = x + \dfrac{1}{x}$, express $x^3 + \dfrac{1}{x^3}$ in terms of $t$.

\item Find the minimum value of $\;3x + \dfrac{27}{x}\;$ for $x > 0$,
      and the value of $x$ at which it is achieved.

\item Solve for real $x$: $\;\sqrt{2x - 1} + \sqrt{x - 1} = 2$.

\item The expression $\;(x+1)(x+3)(x+5)(x+7) + k\;$ is a perfect square for
      all $x$ when $k$ takes a specific value. Find $k$.

\item Prove that $\;(a^2 + b^2)(c^2 + d^2) \geq (ac + bd)^2\;$ for all reals
      $a,b,c,d$. What is the condition for equality?

\item If $u = x + y$ and $v = x - y$, express $x^4 + y^4$ purely in terms of $u$ and $v$.

\item Solve: $\;x^4 - 7x^2 + 12 = 0$.

\end{enumerate}

\section*{Section D \quad Challenge \hfill{\normalfont\small(TMUA / SMC difficulty)}}

\noindent\textit{No brute force. Each question rewards one key observation.}

\begin{enumerate}

\item For all integers $n$, prove that $n^2$ is either divisible by $4$
      or leaves remainder $1$ when divided by $4$.
      Hence prove that $n^2 + 2$ is never divisible by $4$.

\item Find all real solutions to $\;x^4 - 4x^3 + 4x^2 - 1 = 0$.
      (\textit{Look for a way to write the left-hand side as a product of two quadratics.})

\item A sequence is defined by $a_1 = 1$ and $a_{n+1} = \dfrac{a_n}{a_n + 2}$.
      Find a closed form for $a_n$ and hence compute $a_{100}$.

\item Find the value of
\[
  \left(1 - \tfrac{1}{4}\right)\!\left(1 - \tfrac{1}{9}\right)\!\left(1 - \tfrac{1}{16}\right)
    \cdots\left(1 - \tfrac{1}{n^2}\right).
\]
Express your answer in closed form and state its limit as $n \to \infty$.

\item The polynomial $p(x) = x^4 + ax^3 + bx^2 + cx + d$ satisfies $p(1) = 10$,
      $p(2) = 20$, $p(3) = 30$, $p(4) = 40$. Find $p(12) - 12p(0)$ without finding $a,b,c,d$.

\end{enumerate}
