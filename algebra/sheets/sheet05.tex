\SpeedTitleBlock{Daily Algebra Drill \#5}
\noindent\textit{New toolkit today: Partial fractions with quadratic denominators $\cdot$ Elementary functional equations $\cdot$ Cauchy--Schwarz (Engel form) $\cdot$ Conjugate surd pairs revisited.}

\vspace{2pt}
\noindent\textit{Attempt each question before checking answers. Time yourself. No calculators.}

\section*{Section A \quad Rapid Recognition \hfill{\normalfont\small($\leq$15 s each)}}

\noindent\textit{No expansion. Recognise the pattern, write the answer.}

\begin{enumerate}

\item Simplify: $\;\dfrac{x^2 - 9}{x^2 + 5x + 6}$.

\item Rationalise: $\;\dfrac{4}{\sqrt{7} - \sqrt{3}}$.

\item Given $x + y = 3$ and $xy = -4$, find $(x-y)^2$.

\item Factorise: $\;x^3 - x^2 - 4x + 4$.

\item Without expanding, find the coefficient of $x^2$ in $(1+x)^4$.

\item Given $\alpha + \beta = -3$ and $\alpha\beta = -10$, write down the quadratic
      with roots $\alpha$ and $\beta$.

\item Simplify: $\;\sqrt{50} - \sqrt{18} + \sqrt{8}$.

\item Given $a + b = 7$ and $a^2 + b^2 = 29$, find $a^3 + b^3$.

\item Solve: $\;|2x - 3| = 7$.

\item For $x > 0$, find the minimum of $\;\dfrac{(x+4)^2}{x}$.

\end{enumerate}

\section*{Section B \quad Manipulation Drills \hfill{\normalfont\small($\leq$50 s each)}}

\noindent\textit{Fractions, structure, identities. Minimum working.}

\begin{enumerate}

\item Decompose into partial fractions: $\;\dfrac{5x+1}{(x+1)(x-2)}$.

\item Decompose into partial fractions: $\;\dfrac{3}{x(x^2+3)}$.

\item Simplify: $\;\dfrac{1}{\sqrt{n}+\sqrt{n+1}} + \dfrac{1}{\sqrt{n+1}+\sqrt{n+2}}$
      and hence evaluate $\displaystyle\sum_{k=0}^{8}\dfrac{1}{\sqrt{k}+\sqrt{k+1}}$.

\item Factorise: $\;a^4 + a^2 + 1$ over $\mathbb{Z}$.
      (\textit{Hint: add and subtract $a^2$.})

\item Given $a + b = s$ and $ab = p$, express $(a-b)^4$ in terms of $s$ and $p$.

\item Simplify: $\;\dfrac{x^2 - 2x + 1}{x^2 + 2x - 3} \div \dfrac{x^2 - 1}{x^2 + 4x + 3}$.

\item The roots of $x^2 + px + q = 0$ are $r$ and $s$.
      Find in terms of $p$ and $q$ the quadratic with roots $r^2$ and $s^2$.

\item Solve the simultaneous equations:
      $\dfrac{1}{x} + \dfrac{1}{y} = \dfrac{5}{6}$ and $\dfrac{1}{x} - \dfrac{1}{y} = \dfrac{1}{6}$.

\item Prove: $\;\dfrac{1}{1\cdot3}+\dfrac{1}{3\cdot5}+\dfrac{1}{5\cdot7}+\cdots+\dfrac{1}{(2n-1)(2n+1)}
      = \dfrac{n}{2n+1}$.

\item If $f(x) = \dfrac{ax+b}{cx+d}$ and $f(f(x)) = x$ for all $x$,
      what condition must $a, b, c, d$ satisfy?

\end{enumerate}

\section*{Section C \quad Substitution \& Structure \hfill{\normalfont\small($\leq$90 s each)}}

\noindent\textit{One clean substitution is worth ten lines of algebra.}

\begin{enumerate}

\item Solve: $\;\left(x^2 + \dfrac{1}{x^2}\right) - 7\!\left(x - \dfrac{1}{x}\right) - 10 = 0$.

\item For positive reals $a, b, c$ with $a + b + c = 3$, prove
      $\;\dfrac{a^2}{b} + \dfrac{b^2}{c} + \dfrac{c^2}{a} \geq 3\;$
      using the Cauchy--Schwarz (Engel/Titu) form:
      $\dfrac{x^2}{p}+\dfrac{y^2}{q}+\dfrac{z^2}{r}\geq\dfrac{(x+y+z)^2}{p+q+r}$.

\item Solve the system: $\;x^2 + xy = 12,\quad xy + y^2 = 6$.

\item Find the range of values of $k$ for which
      $\;kx^2 - 4x + k - 3 = 0\;$ has two distinct real roots.

\item Simplify $\;\sqrt{3 + 2\sqrt{2}} + \sqrt{3 - 2\sqrt{2}}$ without a calculator.

\item If $\;x - \dfrac{1}{x} = 4$, find the exact value of $\;\sqrt{x} - \dfrac{1}{\sqrt{x}}$.

\item Given $a, b > 0$ with $\dfrac{1}{a} + \dfrac{1}{b} = 1$, show that
      $(a-1)(b-1) = 1$ and find the minimum of $a + b$.

\item Solve: $\;(x^2 - x - 1)^2 = (2x + 1)^2$.

\end{enumerate}

\section*{Section D \quad Challenge \hfill{\normalfont\small(SMC / BMO1 difficulty)}}

\noindent\textit{No brute force. Each question rewards one key observation.}

\begin{enumerate}

\item A function $f$ satisfies $f(x) + f\!\left(\dfrac{1}{1-x}\right) = x$
      for all $x \neq 0, 1$.  Find $f(x)$.
      (\textit{Apply the substitution twice more to get three equations.})

\item For positive reals, prove the Cauchy--Schwarz inequality in \textbf{Engel (Titu) form}:
\[
  \frac{a^2}{x} + \frac{b^2}{y} + \frac{c^2}{z} \;\geq\; \frac{(a+b+c)^2}{x+y+z}.
\]
Hence find the minimum of $\dfrac{1}{x}+\dfrac{4}{1-x}$ for $0<x<1$.

\item Find all functions $f:\mathbb{R}\to\mathbb{R}$ satisfying
      $f(x+y) = f(x) + f(y)$ and $f(1) = 3$.
      (\textit{First find $f$ on $\mathbb{Z}$, then state any assumptions needed for $\mathbb{R}$.})

\item Show that among any 5 integers, there exist at least 2 whose difference is
      divisible by 4.  Hence prove that for any 5 integers, two of them satisfy
      $a^2 \equiv b^2 \pmod{4}$.

\item Solve for integers: $\;x^2 - y^2 = 2024$.

\end{enumerate}
