\SpeedTitleBlock{Daily Algebra Drill \#3}
\noindent\textit{New toolkit today: Remainder \& Factor Theorem $\cdot$ AM--GM \& algebraic inequalities $\cdot$ Completing the square in two variables $\cdot$ Rational Root Theorem $\cdot$ Modular/divisibility arguments $\cdot$ Parametric families of solutions.}

\vspace{2pt}
\noindent\textit{Attempt each question before checking answers. Time yourself. No calculators.}

\section*{Section A \quad Rapid Recognition \hfill{\normalfont\small($\leq$15\,s each)}}
% ═════════════════════════════════════════════════════════════════════════════

\noindent\textit{Zero expansion. Pure pattern recognition.}

\begin{enumerate}

\item The polynomial $f(x)=x^3-3x^2+kx-8$ has $(x-2)$ as a factor. Find $k$.

\item Without dividing, find the remainder when $p(x)=x^{100}-1$ is divided by $(x-1)$.

\item Factorise: $\;6x^2-x-12$.

\item Simplify: $\;\dfrac{x^{2n}-1}{x^n-1}\qquad(n\in\mathbb{Z}^+)$.

\item Given $x>0$, find the minimum value of $\;x+\dfrac{9}{x}$.

\item Factorise over $\mathbb{Z}$: $\;x^4-y^4$.

\item The roots of $2x^2-6x+3=0$ are $\alpha$ and $\beta$. Find $\alpha^2+\beta^2$.

\item Simplify: $\;\dfrac{(n+1)!-n!}{(n-1)!}\qquad(n\geq 1)$.

\item Without expanding, show that $(a+b+c)^2-(a^2+b^2+c^2)=2(ab+bc+ca)$
and state the value of $ab+bc+ca$ when $a+b+c=4$ and $a^2+b^2+c^2=8$.

\item Find the value of $k$ such that $x^2+kx+16$ is a perfect square.

\end{enumerate}

% ═════════════════════════════════════════════════════════════════════════════
\section*{Section B \quad Manipulation Drills \hfill{\normalfont\small($\leq$50\,s each)}}
% ═════════════════════════════════════════════════════════════════════════════

\noindent\textit{Fractions, rearranging, hidden factorisations. Minimal expansion.}

\begin{enumerate}

\item Find all rational roots of $\;2x^3-3x^2-11x+6$, and hence factorise it completely.

\item Simplify: $\;\dfrac{x^3+1}{x^2-x+1}$.

\item Given $a+b=p$ and $a^3+b^3=q$, express $ab$ in terms of $p$ and $q$.

\item For $x\neq0$, simplify: $\;\dfrac{x^2(x+1)-x(x+1)^2}{x(x+1)}$.

\item Factorise: $\;a^3-a^2b-ab^2+b^3$.

\item The sum $\displaystyle\sum_{r=1}^{n}r = \dfrac{n(n+1)}{2}$.
      Use this to find, without expanding, $\displaystyle\sum_{r=1}^{n}(2r-1)$.

\item Solve for $x$: $\;\dfrac{2x-1}{x+3}-\dfrac{x+2}{x-1}=1$.

\item Simplify: $\;\left(\dfrac{1}{a}-\dfrac{1}{b}\right)\div\left(\dfrac{1}{a^2}-\dfrac{1}{b^2}\right)$.

\item Show that $n^3-n$ is divisible by 6 for all integers $n$.

\item Given $\alpha,\beta$ are roots of $x^2-x-1=0$, find $\alpha^3+\beta^3$
      and $\alpha^4+\beta^4$ without solving for $\alpha$ or $\beta$.

\end{enumerate}

% ═════════════════════════════════════════════════════════════════════════════
\section*{Section C \quad Substitution \& Structure \hfill{\normalfont\small($\leq$90\,s each)}}
% ═════════════════════════════════════════════════════════════════════════════

\noindent\textit{Find structure before computing. One good substitution beats ten lines of algebra.}

\begin{enumerate}

\item Solve: $\;9^x - 10\cdot 3^x + 9 = 0$.

\item Find the minimum value of $\;x^2+y^2\;$ subject to $\;x+2y=5$.

\item Prove that $\;a^2+b^2+c^2 \geq ab+bc+ca\;$ for all real $a,b,c$.

\item Solve: $\;\sqrt{x+5}+\sqrt{x-3}=4$.

\item The expression $x^2+4x+y^2-6y+13$ has a minimum value.
      Find it, and the values of $x$ and $y$ at which it is achieved.

\item If $x+\dfrac{1}{x}=5$, find $\dfrac{x^4+1}{x^2}$.

\item Solve over the reals: $\;x^4+4x^2-5=0$.

\item Given $a,b>0$ with $a+b=1$, find the minimum of $\;\dfrac{1}{a}+\dfrac{1}{b}$.

\end{enumerate}

% ═════════════════════════════════════════════════════════════════════════════
\section*{Section D \quad Challenge \hfill{\normalfont\small(TMUA / SMC difficulty)}}
% ═════════════════════════════════════════════════════════════════════════════

\noindent\textit{Five lines or fewer. Find the structure, not the computation.}

\begin{enumerate}

\item Find all real solutions to $\;x^4+2x^3-2x-1=0$.
      (\textit{Hint: the left-hand side factorises into two quadratics.})

\item Prove that for all integers $n$, the expression $n^2+n+41$ is not always prime
      by finding the smallest counterexample. Then explain why $n^2+n+41$ is prime
      for $n=0,1,2,\ldots,39$.

\item Let $a,b,c>0$ with $abc=1$. Prove that $\;a+b+c\geq3$.

\item For positive reals $a$ and $b$, prove that
\[
  \frac{a}{b}+\frac{b}{a}\;\geq\;2.
\]
Hence prove the AM--GM inequality $\;\dfrac{a+b}{2}\geq\sqrt{ab}\;$ for $a,b>0$.

\item Find the number of integers $n$ with $1\leq n\leq 1000$ such that $n^2-1$ is
      divisible by $n+3$.

\end{enumerate}
