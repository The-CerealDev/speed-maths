\section*{Section A \quad Rapid Recognition \hfill{\normalfont\small($\leq$15\,s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Negate a statement, or state a contrapositive. Pure recognition.}

\begin{enumerate}

\item Negate: ``$n$ is even.''

\item Negate: ``$x>5$.''

\item Negate: ``$n$ is divisible by $3$ and by $5$.''

\item Negate: ``$x=2$ or $x=3$.''

\item Negate: ``For all integers $n$, $n^2 \geq 0$.''

\item Negate: ``There exists an integer $n$ such that $n^2=2$.''

\item True or false: ``The negation of a true statement is always false.''

\item State the contrapositive of: ``If $n$ is even, then $n^2$ is even.''

\item True or false: ``The contrapositive of a statement is always logically equivalent to the original.''

\item True or false: ``The contrapositive of a statement is the same as its converse.''

\end{enumerate}

\section*{Section B \quad Manipulation Drills \hfill{\normalfont\small($\leq$50\,s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Negate an implication or a compound statement; build your first contrapositive proof.}

\begin{enumerate}

\item Negate: ``If $n$ is prime, then $n$ is odd.''

\item Negate: ``For all $x>0$, $x^2>0$.''

\item Negate: ``$n$ is a multiple of $4$ or $n$ is a multiple of $6$.''

\item State the contrapositive of: ``If $x>2$, then $x^2>4$.''

\item Prove ``If $n^2$ is odd, then $n$ is odd'' by proving its contrapositive.

\item Negate: ``$n$ is prime and $n>2$.''

\item Negate: ``If it rains, then the ground is wet.''

\item State the contrapositive of: ``If $n$ is a multiple of $9$, then $n$ is a multiple of $3$.''

\item True or false: ``$\text{not}(P \text{ and } Q)$'' is the same as ``$(\text{not } P) \text{ and } (\text{not } Q)$''.

\item True or false: ``$\text{not}(P \text{ or } Q)$'' is the same as ``$(\text{not } P) \text{ and } (\text{not } Q)$''.

\end{enumerate}

\section*{Section C \quad Substitution \& Structure \hfill{\normalfont\small($\leq$90\,s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Negate quantified statements, prove via contrapositive, and spot a De Morgan trap.}

\begin{enumerate}

\item Negate: ``For every prime $p$, $p$ is odd or $p=2$.''

\item Prove ``If $n^2$ is even, then $n$ is even'' via its contrapositive.

\item Which is the correct negation of ``If $x$ is rational, then $x^2$ is rational''?
\begin{itemize}
\item[A)] If $x$ is irrational, then $x^2$ is irrational.
\item[B)] $x$ is rational and $x^2$ is irrational.
\item[C)] If $x^2$ is irrational, then $x$ is irrational.
\item[D)] $x$ is irrational and $x^2$ is rational.
\end{itemize}

\item Negate: ``There exists an integer $n$ such that $n$ is prime and $n$ is even.''

\item State and prove the contrapositive of: ``If $n$ is a multiple of $4$, then $n^2$ is a multiple of $16$.''

\item True or false, with justification: ``If a statement's contrapositive is false, the original statement must also be false.''

\item Negate: ``There exists a positive integer $n$ such that $n$ is not a multiple of any prime less than $10$.''

\item Prove ``If $n$ is not a multiple of $3$, then $n^2$ is not a multiple of $3$'' via its contrapositive.

\end{enumerate}

\section*{Section D \quad Challenge \hfill{\normalfont\small(TMUA / SMC difficulty)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Insight over computation. Each question has a short, elegant solution --- find it.}

\begin{enumerate}

\item Given the statement $S$: ``If $n$ is a multiple of $4$, then $n$ is even.'' Which of the following is the contrapositive of $S$?
\begin{itemize}
\item[A)] If $n$ is not even, then $n$ is not a multiple of $4$.
\item[B)] If $n$ is even, then $n$ is a multiple of $4$.
\item[C)] If $n$ is not a multiple of $4$, then $n$ is not even.
\item[D)] If $n$ is odd, then $n$ is a multiple of $4$.
\end{itemize}

\item Prove by contrapositive: ``If $n^2 - 2n$ is odd, then $n$ is odd.''

\item On an island, every inhabitant either always tells the truth or always lies. You meet two people, $A$ and $B$. $A$ says: ``At least one of us is a liar.'' Determine, with justification, whether $A$ and $B$ are truth-tellers or liars.

\item Determine, with proof via contrapositive, whether the following is true: ``If $n^3$ is even, then $n$ is even.''

\item Statement $S$: ``If $n$ is a multiple of $15$, then $n$ is a multiple of $3$ and a multiple of $5$.'' (a) State the contrapositive of $S$. (b) Is ``$n$ is a multiple of $15$'' necessary and sufficient for ``$n$ is a multiple of $3$ and a multiple of $5$''? (c) Negate $S$ directly (without first simplifying via contrapositive).

\end{enumerate}
