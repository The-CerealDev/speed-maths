\SpeedTitleBlock{Daily Logic Drill \#4}{David Akinyele-Aje}
\noindent\textit{New toolkit today: Disproving Universal Claims $\cdot$ Minimal Counterexamples $\cdot$ Common Counterexample Sources ($0$, $1$, $-1$, non-integers).}

\vspace{2pt}
\noindent\textit{Attempt each question before checking answers. Time yourself. No calculators.}

\section*{Section A \quad Rapid Recognition \hfill{\normalfont\small($\leq$15\,s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Kill a universal claim with a single well-chosen example. Pure recognition.}

\begin{enumerate}

\item Give a counterexample to: ``Every prime number is odd.''

\item Give a counterexample to: ``$n^2 \geq n$ for every real number $n$.''

\item Give a counterexample to: ``If $n$ is prime, then $n+2$ is prime.''

\item Give a counterexample to: ``The sum of two primes is always even.''

\item Give a counterexample to: ``Every multiple of $4$ is a multiple of $8$.''

\item Give a counterexample to: ``$\sqrt{a+b} = \sqrt{a}+\sqrt{b}$ for all $a,b \geq 0$.''

\item Give a counterexample to: ``$(x+y)^2 = x^2+y^2$ for all real $x,y$.''

\item Give a counterexample to: ``$n!+1$ is always prime.''

\item Give a counterexample to: ``If a number is not prime, it must be even.''

\item Give a counterexample to: ``All odd numbers greater than $1$ are prime.''

\end{enumerate}

\section*{Section B \quad Manipulation Drills \hfill{\normalfont\small($\leq$50\,s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Hunt systematically, and notice where $0$, $1$, $-1$, and non-integers do the most damage.}

\begin{enumerate}

\item Give a counterexample to: ``$n^2+n+41$ is prime for every positive integer $n$.''

\item Give a counterexample to: ``If $a$ and $b$ are both irrational, then $a+b$ is irrational.''

\item Give a counterexample to: ``If $a$ and $b$ are both irrational, then $ab$ is irrational.''

\item Give a counterexample to: ``Every quadratic with integer coefficients that has an integer root must have two integer roots.''

\item Give a counterexample (using $n=0$) to: ``For every integer $n$, $n/n = 1$.''

\item Give a counterexample (using $n=1$) to: ``Every positive integer has at least one prime factor.''

\item Give a counterexample (using $n=-1$) to: ``For all integers $n$, $n^n$ is positive.''

\item Give a counterexample (using a non-integer) to: ``If $x^2$ is an integer, then $x$ is an integer.''

\item Give a counterexample to: ``If $p$ and $q$ are distinct primes, then $p+q$ is composite.''

\item Give a counterexample to: ``For every integer $n \geq 1$, $2^n - 1$ is prime.''

\end{enumerate}

% ══════════════════════════════════════════════════════════════════════════════
\section*{Section C \quad Substitution \& Structure \hfill{\normalfont\small($\leq$90\,s each)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Find the smallest counterexample, not just any counterexample --- and spot a structural trap.}

\begin{enumerate}

\item Find the smallest positive integer $n$ for which $n^2+n+41$ is \emph{not} prime.

\item Give a counterexample to: ``If $a \equiv b \pmod n$, then $a^2 \equiv b^2 \pmod{n^2}$.''

\item Which of the following values of $n$ is a counterexample to ``$n^2-n+11$ is prime for every positive integer $n$''?
\begin{itemize}
\item[A)] $n=1$
\item[B)] $n=5$
\item[C)] $n=11$
\item[D)] $n=10$
\end{itemize}

\item Give a counterexample to: ``If $p$ is prime, then $2^p-1$ is prime.''

\item Give a counterexample to: ``If $a^2 \equiv b^2 \pmod n$, then $a \equiv b \pmod n$.''

\item Find the smallest positive integer $n$ for which $3^n+2$ is \emph{not} prime.

\item Give a counterexample to: ``If $x$ and $y$ are both non-integers, then $x+y$ is a non-integer.''

\item Give a counterexample to: ``For all integers $a,b,c$: if $a \mid bc$, then $a \mid b$ or $a \mid c$.''

\end{enumerate}

\section*{Section D \quad Challenge \hfill{\normalfont\small(TMUA / SMC difficulty)}}
% ══════════════════════════════════════════════════════════════════════════════

\noindent\textit{Insight over computation. Each question has a short, elegant solution --- find it.}

\begin{enumerate}

\item Consider the claims: I. ``$n^2+n+41$ is prime for every positive integer $n$.'' II. ``$2^n-1$ is prime whenever $n$ is prime.'' III. ``$n!+1$ is prime for every positive integer $n$.'' Which of I, II, III are true?
\begin{itemize}
\item[A)] I only
\item[B)] II only
\item[C)] III only
\item[D)] I and II only
\item[E)] I and III only
\item[F)] II and III only
\item[G)] I, II and III
\item[H)] None of them
\end{itemize}

\item Find the smallest positive integer $n$ for which $n^2+n+1$ is \emph{not} prime.

\item Give a counterexample to: ``The sum of any three consecutive prime numbers is always even.''

\item Give a counterexample to: ``If $n$ is a positive integer such that $n^2+1$ is prime, then $n$ must be even.''

\item Statement $T$ (you may take this as given/true): ``If $2^n-1$ is prime, then $n$ is prime.'' (a) Explain why B10 and C4's counterexamples do \emph{not} contradict $T$. (b) State $T$'s converse, and determine whether it is true. (c) Is ``$2^n-1$ is prime'' necessary and sufficient for ``$n$ is prime''?

\end{enumerate}
